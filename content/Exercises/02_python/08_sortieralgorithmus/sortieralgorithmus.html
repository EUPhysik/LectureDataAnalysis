def list_str(A, n=10):
    res = ""
    res += "["
    nflines = len(A)//n
    nhlines = len(A)%n
    for i in range(nflines):
        if nflines > 0:
            res += " "
        for j in range(n):
            res += str(A[i*n+j])
            if i*n+j < len(A)-1:
                res += ", "
        if not (nhlines == 0 and i == nflines-1):
            res += "\n"
    for j in range(nhlines):
        res += str(A[nflines*n+j])
        if j < nhlines-1:
                res += ", "
    res += "]"
    return res

def bubble_sort(A, ausgabe=False):

    # Initialer Wert f√ºr n, wird bei jedem Schleifendurchlauf 
    # um Eins verringert
    n = len(A)

    # Variable zum Pr√ºfen ob eine Vertauschung erfolgt ist 
    # initial wird sie auf wahr gesetzt
    getauscht = True
    
    # Iterationsz√§hler
    izaehler = 0
    
    while getauscht == True:
        getauscht = False
        izaehler += 1
        
        if ausgabe == True:
            print("Iteration:", izaehler)
            print("   Liste zu Beginn der Iteration: ")
            print("   ", list_str(A))
        
        # Vertauschungsschleife
        for j in range(n-1):
            # Pr√ºfen ob das nachfolgende Element kleiner ist,
            # falls ja, werden beide getauscht
            if A[j+1] < A[j]:
                if ausgabe == True:
                    print("   Tausche: ", A[j], "und", A[j+1])
                
                # Tauschen des j mit dem j+1 Elementwert
                mv = A[j]
                A[j] = A[j+1]
                A[j+1] = mv
                
                # Markierung, dass eine Vertauschung stattfand
                getauscht = True
                
                if ausgabe == True:
                    print("   Liste nach Tausch: ")
                    print("   ", list_str(A))
            else:
                if ausgabe == True:
                    print("   Elemente ", A[j], "und", A[j+1], " m√ºssen nicht getauscht werden")

        # Schleifendurchlauf wird verk√ºrzt 
        n -= 1
        if not getauscht:
            if ausgabe == True:
                print("   kein Tausch mehr notwendig, Liste ist nun sortiert")
    return A

def bubble_sort_zaehler(A):

    # Initialer Wert f√ºr n, wird bei jedem Schleifendurchlauf 
    # um Eins verringert
    n = len(A)
    
    # Variable zum Pr√ºfen ob eine Vertauschung erfolgt ist 
    # initial wird sie auf wahr gesetzt
    getauscht = True
    
    # Z√§hler f√ºr Vertauschungen
    vertauschungen = 0
    
    # Schleife wird solgange ausgef√ºhrt bis keine
    # weitere Vertauschung notwendig ist
    while getauscht == True:
        getauscht = False
        
        # Vertauschungsschleife
        for j in range(n-1):
            # Pr√ºfen ob das nachfolgende Element kleiner ist,
            # falls ja, werden beide getauscht
            if A[j+1] < A[j]:

                # Tauschen des j mit dem j+1 Elementwert
                mv = A[j]
                A[j] = A[j+1]
                A[j+1] = mv
                
                # Markierung, dass eine Vertauschung stattfand
                getauscht = True
                
                # Erh√∂he Z√§hler der Vertauschungen
                vertauschungen += 1
                
        # Schleifendurchlauf wird verk√ºrzt 
        n -= 1

    # R√ºckgabe der sortierten Liste als auch der Anzahl 
    # der Vertauschungen 
    return A, vertauschungen
